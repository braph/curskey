122 ( \
123  ((MOD) & ~(CURSKEY_MOD_CTRL|CURSKEY_MOD_META|CURSKEY_MOD_SHIFT)) ? ( \
124  ERR /* Invalid modifier */ \
125  ) \
126  :(((KEY) >= 'a' && (KEY) <= 'z') || ((KEY) >= 'A' && (KEY) <= 'Z')) ? ( \
127  (((KEY) \
128  & (((MOD) & CURSKEY_MOD_SHIFT) ? ~0x20 : 0xFF)) \
129  % (((MOD) & CURSKEY_MOD_CTRL) ? 32 : 0xFF)) \
130  + (((MOD) & CURSKEY_MOD_META) ? CURSKEY_META_START : 0) \
131  ) \
